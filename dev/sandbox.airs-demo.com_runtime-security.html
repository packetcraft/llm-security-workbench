<pre id="complete-code">import os
import aisecurity
from aisecurity.scan.inline.scanner import Scanner
from aisecurity.generated_openapi_client.models.ai_profile import AiProfile
from aisecurity.scan.models.content import Content

# Initialize once at app startup
aisecurity.init(api_key=os.getenv("PANW_AI_SEC_API_KEY"))
scanner = Scanner()
ai_profile = AiProfile(profile_name="your-security-profile")

def process_user_query(user_id, user_prompt):
"""
Secure AI query processing with dual-phase scanning
"""
# PHASE 1: Scan user input
input_scan = scanner.sync_scan(
ai_profile=ai_profile,
content=Content(prompt=user_prompt),
metadata={"app_user": user_id}
)

if input_scan.action == "block":
return "Your request violates security policies."

# Process with your AI model (OpenAI, Anthropic, etc.)
ai_response = your_ai_model(user_prompt)

# PHASE 2: Scan AI response
output_scan = scanner.sync_scan(
ai_profile=ai_profile,
content=Content(prompt=user_prompt, response=ai_response),
metadata={"app_user": user_id}
)

if output_scan.action == "block":
return "I cannot provide that information."

return ai_response

# Example usage
result = process_user_query("user_123", "Tell me about your products")</pre>

function getOrCreateSessionId() {
let sessionId = sessionStorage.getItem('runtime_session_id');
if (!sessionId) {
sessionId = 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
sessionStorage.setItem('runtime_session_id', sessionId);
}
return sessionId;
}

function updateAirsSessionLink() {
const sessionId = getOrCreateSessionId();
const airsLink = document.getElementById('airs-session-link');
if (airsLink) {
const airsUrl = `https://stratacloudmanager.paloaltonetworks.com/ai-security/runtime/ai-sessions/${sessionId}/6196b46a-08cf-4b56-9840-fa02d8dad240/AIRS-sandbox?tsg_id=1135250260`;
airsLink.href = airsUrl;
}
}

function startNewSession() {
// Clear conversation history
conversationHistory = [];

// Remove old session ID from storage
sessionStorage.removeItem('runtime_session_id');

const messagesContainer = document.getElementById('chatbot-messages');
if (messagesContainer) {
messagesContainer.innerHTML = `
<div class="message bot">
<div class="bot-message-content">
Hi! I'm your Runtime Security chatbot, powered by Prisma AIRS. Try asking me questions or select a sample threat to test security blocking.
</div>
</div>
`;
}

// Reset scan results panels
const inputScanContent = document.getElementById('input-scan-content');

}

systemMessage.className = 'message system';
systemMessage.textContent = '🔄 New session started';
messagesContainer.appendChild(systemMessage);
scrollToBottom();
}

function handleSecurityToggle() {
const toggle = document.getElementById('security-toggle');
securityEnabled = toggle.checked;

// Update status indicator
const statusIndicator = document.getElementById('security-toggle-status');
if (statusIndicator) {
if (securityEnabled) {
statusIndicator.textContent = 'ON';
statusIndicator.className = 'status-indicator status-enabled';
} else {
statusIndicator.textContent = 'OFF';
statusIndicator.className = 'status-indicator status-disabled';
}
}

// Update scan results display
const inputScanContent = document.getElementById('input-scan-content');
const outputScanContent = document.getElementById('output-scan-content');

if (!securityEnabled) {
// Show "disabled" message when scanning is off
if (inputScanContent) {
inputScanContent.innerHTML = '<div class="scan-disabled-message"><i class="fas fa-shield-slash"></i> Security scanning disabled</div>';
}
if (outputScanContent) {
outputScanContent.innerHTML = '<div class="scan-disabled-message"><i class="fas fa-shield-slash"></i> Security scanning disabled</div>';
}
} else {
// Show "no data" message when scanning is on but no scans yet
if (inputScanContent) {
inputScanContent.innerHTML = '<div class="no-scan-data">No scan data available</div>';
}
if (outputScanContent) {
outputScanContent.innerHTML = '<div class="no-scan-data">No scan data available</div>';
}
}

// Add system message
const messagesContainer = document.getElementById('chatbot-messages');
const systemMessage = document.createElement('div');
systemMessage.className = 'message system';
systemMessage.textContent = securityEnabled
? '🛡️ Security scanning enabled'
: '⚠️ Security scanning disabled';
messagesContainer.appendChild(systemMessage);
scrollToBottom();
}

function syntaxHighlightJSON(json) {
if (typeof json !== 'string') {
json = JSON.stringify(json, null, 2);
}
json = json.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
return json.replace(/("(\\u[a-zA-Z0-9]{4}|\\[^u]|[^\\"])*"(\s*:)?|\b(true|false|null)\b|-?\d+(?:\.\d*)?(?:[eE][+\-]?\d+)?)/g, function (match) {
let cls = 'json-number';
if (/^"/.test(match)) {
if (/:$/.test(match)) {
cls = 'json-key';
} else {
cls = 'json-string';
}
} else if (/true|false/.test(match)) {
cls = 'json-boolean';
} else if (/null/.test(match)) {
cls = 'json-null';
}
return '<span class="' + cls + '">' + match + '</span>';
});
}

// Auth error detection helpers
function isAuthError(response, error) {
// Check for redirect to OAuth server
if (response && response.redirected && response.url.includes('oauth')) {
return true;
}
// Check for auth error status codes
if (response && (response.status === 401 || response.status === 403)) {
return true;
}
// CORS errors from OAuth redirect appear as TypeError: Failed to fetch
if (error instanceof TypeError && error.message.includes('Failed to fetch')) {
return true;
}
return false;
}

function handleSessionExpired() {
removeTypingIndicator();
addBotMessage("Your session has expired. The page will reload to re-authenticate.");
setTimeout(() => window.location.reload(), 2000);
}

function updateScanResults(securityInfo) {
const inputScanContent = document.getElementById('input-scan-content');
const outputScanContent = document.getElementById('output-scan-content');

if (!securityInfo) {
inputScanContent.innerHTML = '<div class="no-scan-data">No scan data available</div>';
outputScanContent.innerHTML = '<div class="no-scan-data">No scan data available</div>';
return;
}

// Update input scan
if (securityInfo.input_scan) {
const highlightedJSON = syntaxHighlightJSON(securityInfo.input_scan);
inputScanContent.innerHTML = '<pre>' + highlightedJSON + '</pre>';
} else {
inputScanContent.innerHTML = '<div class="no-scan-data">No input scan data</div>';
}

// Update output scan
if (securityInfo.output_scan) {
const highlightedJSON = syntaxHighlightJSON(securityInfo.output_scan);
outputScanContent.innerHTML = '<pre>' + highlightedJSON + '</pre>';
} else {
outputScanContent.innerHTML = '<div class="no-scan-data">No output scan data</div>';
}
}

async function sendChatbotMessage() {
const input = document.getElementById('chatbot-input');
const message = input.value.trim();

if (!message) return;

addUserMessage(message);
input.value = '';

conversationHistory.push({
role: 'user',
content: message
});

showTypingIndicator();

try {
const response = await fetch('/runtime-security/api/chat', {
method: 'POST',
headers: {
'Content-Type': 'application/json',
},
body: JSON.stringify({
message: message,
session_id: getOrCreateSessionId(),
security_enabled: securityEnabled,
context: {
site_id: "runtime_security",
type: "runtime_protection_testing"
}
})
});

// Check for auth redirect (session expired)
if (isAuthError(response, null)) {
handleSessionExpired();
return;
}

if (!response.ok) {
throw new Error(`HTTP error! status: ${response.status}`);
}

const data = await response.json();
removeTypingIndicator();

// Update scan results panel
if (data.security_info) {
updateScanResults(data.security_info);
} else if (!securityEnabled) {
// Show disabled message
const inputScanContent = document.getElementById('input-scan-content');
const outputScanContent = document.getElementById('output-scan-content');
if (inputScanContent) {
inputScanContent.innerHTML = '<div class="no-scan-data">Security scanning disabled</div>';
}
if (outputScanContent) {
outputScanContent.innerHTML = '<div class="no-scan-data">Security scanning disabled</div>';
}
}

if (data.response) {
addBotMessage(data.response);
conversationHistory.push({
role: 'assistant',
content: data.response
});
}

} catch (error) {
console.error('Error:', error);
// Check if this is an auth error (CORS failure from OAuth redirect)
if (isAuthError(null, error)) {
handleSessionExpired();
return;
}
removeTypingIndicator();
addBotMessage("I apologize, but I'm experiencing technical difficulties. Please try again.");
}
}

function addUserMessage(message) {
const messagesContainer = document.getElementById('chatbot-messages');
const messageDiv = document.createElement('div');
messageDiv.className = 'message user';
messageDiv.textContent = message;
messagesContainer.appendChild(messageDiv);
scrollToBottom();
}

function formatBotMessage(text) {
text = text.replace(/^[\(\)\[\]\{\}\s]+/gm, '');
text = text.replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>');
text = text.replace(/\\n/g, '\n');
text = text.replace(/\n/g, '<br>');
text = text.replace(/(^|<br>)\*\s+/g, '$1• ');
text = text.replace(/(^|<br>)[-+]\s+/g, '$1• ');
text = text.replace(/(<br>){3,}/g, '<br><br>');
return text.trim();
}

function addBotMessage(message) {
const messagesContainer = document.getElementById('chatbot-messages');
const messageDiv = document.createElement('div');
messageDiv.className = 'message bot';
const formattedMessage = formatBotMessage(message);
messageDiv.innerHTML = `<div class="bot-message-content">${formattedMessage}</div>`;
messagesContainer.appendChild(messageDiv);
scrollToBottom();
}

// Populate the threat library sidebar with categorized threats
function populateThreatLibrary() {
const container = document.getElementById('threat-library-content');
if (!container) return;

// Clear existing content
container.innerHTML = '';

// Calculate total threat count
let totalThreats = 0;
SAMPLE_THREATS.forEach(category => {
totalThreats += category.threats.length;
});

// Update threat count display
const threatCountDisplay = document.getElementById('threat-count-display');
if (threatCountDisplay) {
threatCountDisplay.textContent = `${totalThreats} Threats Available`;
}

// Create category sections
SAMPLE_THREATS.forEach((category, categoryIndex) => {
// Category container
const categoryDiv = document.createElement('div');
categoryDiv.className = 'threat-category';
categoryDiv.id = `category-${category.categoryId}`;

// Category header
const headerDiv = document.createElement('div');
headerDiv.className = 'threat-category-header';
headerDiv.onclick = () => toggleCategory(category.categoryId);

headerDiv.innerHTML = `
<div class="threat-category-info">
<span class="threat-category-name">${category.category}</span>
<span class="threat-category-description">${category.description}</span>
</div>
<i class="fas fa-chevron-down threat-category-toggle"></i>
`;

categoryDiv.appendChild(headerDiv);

// Threat list
const threatListDiv = document.createElement('div');
threatListDiv.className = 'threat-list';

category.threats.forEach((threat, threatIndex) => {
const threatItem = document.createElement('div');
threatItem.className = 'threat-item';
threatItem.onclick = () => handleThreatClick(categoryIndex, threatIndex);

threatItem.innerHTML = `
<i class="fas fa-angle-right"></i>
<span class="threat-item-name">${threat.type}</span>
`;

threatListDiv.appendChild(threatItem);
});

categoryDiv.appendChild(threatListDiv);
container.appendChild(categoryDiv);
});
}

// Toggle category expansion/collapse
function toggleCategory(categoryId) {
const category = document.getElementById(`category-${categoryId}`);
if (!category) return;

category.classList.toggle('collapsed');
}

// Handle threat item click
function handleThreatClick(categoryIndex, threatIndex) {
const input = document.getElementById('chatbot-input');
if (!input) return;

const threat = SAMPLE_THREATS[categoryIndex].threats[threatIndex];
if (!threat) return;

// Fill input field with threat example
input.value = threat.example;
input.focus();

// Close sidebar on mobile after selection
if (window.innerWidth <= 768) {
toggleThreatLibrary();
}

// Visual feedback: Briefly highlight the selected item
const allItems = document.querySelectorAll('.threat-item');
const selectedItem = allItems[
SAMPLE_THREATS.slice(0, categoryIndex).reduce((sum, cat) => sum + cat.threats.length, 0) + threatIndex
];

if (selectedItem) {
selectedItem.style.background = 'rgba(0, 176, 255, 0.3)';
setTimeout(() => {
selectedItem.style.background = '';
}, 500);
}
}

// Toggle threat library sidebar (mobile)
function toggleThreatLibrary() {
const sidebar = document.querySelector('.column-left');
const overlay = document.getElementById('threat-library-overlay');

if (!sidebar || !overlay) return;

sidebar.classList.toggle('active');
overlay.classList.toggle('active');

// Prevent body scroll when sidebar is open on mobile
if (sidebar.classList.contains('active')) {
document.body.style.overflow = 'hidden';
} else {
document.body.style.overflow = 'auto';
}
}
